163 template<typename Policy = as_is_t, typename Type>
164 meta_any meta_dispatch([[maybe_unused]] Type &&value) {
165  if constexpr(std::is_same_v<Policy, as_void_t>) {
166  return meta_any{std::in_place_type<void>};
167  } else if constexpr(std::is_same_v<Policy, as_ref_t>) {
168  return meta_any{std::in_place_type<Type>, std::forward<Type>(value)};
169  } else if constexpr(std::is_same_v<Policy, as_cref_t>) {
170  static_assert(std::is_lvalue_reference_v<Type>, "Invalid type");
171  return meta_any{std::in_place_type<const std::remove_reference_t<Type> &>, std::as_const(value)};
172  } else {
173  static_assert(std::is_same_v<Policy, as_is_t>, "Policy not supported");
174  return meta_any{std::forward<Type>(value)};
175  }
