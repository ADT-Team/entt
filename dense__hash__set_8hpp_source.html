277  template<typename Arg>
278  [[nodiscard]] auto get_or_emplace(Arg &&arg) {
279  const auto hash = sparse.second()(arg);
280  auto index = hash_to_bucket(hash);
281 
282  if(auto it = constrained_find(arg, index); it != end()) {
283  return std::make_pair(it, false);
284  }
285 
286  if(const auto count = size() + 1u; count > (bucket_count() * max_load_factor())) {
287  rehash(bucket_count() * 2u);
288  index = hash_to_bucket(hash);
289  }
290 
291  packed.first().emplace_back(sparse.first()[index], std::forward<Arg>(arg));
292  // update goes after emplace to enforce exception guarantees
293  sparse.first()[index] = size() - 1u;
294 
295  return std::make_pair(--end(), true);
296  }

298  template<typename Other>
299  bool do_erase(const Other &value) {
300  for(size_type *curr = sparse.first().data() + bucket(value); *curr != std::numeric_limits<size_type>::max(); curr = &packed.first()[*curr].next) {
301  if(packed.second()(packed.first()[*curr].element, value)) {
302  const auto index = *curr;
303  *curr = packed.first()[*curr].next;
304  move_and_pop(index);
305  return true;
306  }
307  }
308 
309  return false;
310  }

312  void move_and_pop(const std::size_t pos) {
313  if(const auto last = size() - 1u; pos != last) {
314  size_type *curr = sparse.first().data() + bucket(packed.first().back().element);
315  for(; *curr != last; curr = &packed.first()[*curr].next) {}
316  *curr = pos;
317 
318  // basic exception guarantees when value type has a throwing move constructor
319  packed.first()[pos] = std::move(packed.first().back());
320  }
321 
322  packed.first().pop_back();
323  }

800  void rehash(const size_type count) {
801  auto value = (std::max)(count, minimum_capacity);
802  value = (std::max)(value, static_cast<size_type>(size() / max_load_factor()));
803 
804  if(const auto sz = next_power_of_two(value); sz != bucket_count()) {
805  sparse.first().resize(sz);
806  std::fill(sparse.first().begin(), sparse.first().end(), std::numeric_limits<size_type>::max());
807 
808  for(size_type pos{}, last = size(); pos < last; ++pos) {
809  const auto index = bucket(packed.first()[pos].element);
810  packed.first()[pos].next = std::exchange(sparse.first()[index], pos);
811  }
812  }
