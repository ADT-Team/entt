248 [[nodiscard]] adjacency_matrix<directed_tag> graph() const {
249 const auto length = vertices.size();
250 adjacency_matrix<directed_tag> matrix{length};
251
252 // creates the adjacency matrix
253 for(const auto &elem: deps) {
254 const auto last = elem.second.cend();
255 auto it = elem.second.cbegin();
256
257 while(it != last) {
258 if(it->second) {
259 // rw item
260 if(auto curr = it++; it != last) {
261 if(it->second) {
262 matrix.insert(curr->first, it->first);
263 } else if(const auto next = std::find_if(it, last, [](const auto &value) { return value.second; }); next != last) {
264 for(; it != next; ++it) {
265 matrix.insert(curr->first, it->first);
266 matrix.insert(it->first, next->first);
267 }
268 } else {
269 for(; it != next; ++it) {
270 matrix.insert(curr->first, it->first);
271 }
272 }
273 }
274 } else {
275 // ro item (first iteration only)
276 if(const auto next = std::find_if(it, last, [](const auto &value) { return value.second; }); next != last) {
277 for(; it != next; ++it) {
278 matrix.insert(it->first, next->first);
279 }
280 } else {
281 it = last;
282 }
283 }
284 }
285 }
286
287 // computes the transitive closure
288 for(std::size_t vk{}; vk < length; ++vk) {
289 for(std::size_t vi{}; vi < length; ++vi) {
290 for(std::size_t vj{}; vj < length; ++vj) {
291 if(matrix.contains(vi, vk) && matrix.contains(vk, vj)) {
292 matrix.insert(vi, vj);
293 }
294 }
295 }
296 }
297
298 // applies the transitive reduction
299 for(std::size_t vert{}; vert < length; ++vert) {
300 matrix.erase(vert, vert);
301 }
302
303 for(std::size_t vj{}; vj < length; ++vj) {
304 for(std::size_t vi{}; vi < length; ++vi) {
305 if(matrix.contains(vi, vj)) {
306 for(std::size_t vk{}; vk < length; ++vk) {
307 if(matrix.contains(vj, vk)) {
308 matrix.erase(vi, vk);
309 }
310 }
311 }
312 }
313 }
314
315 return matrix;
316 }
