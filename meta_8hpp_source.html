158  static void basic_vtable([[maybe_unused]] const operation op, [[maybe_unused]] const any &value, [[maybe_unused]] void *other) {
159  static_assert(std::is_same_v<std::remove_reference_t<std::remove_const_t<Type>>, Type>, "Invalid type");
160 
161  if constexpr(!std::is_void_v<Type>) {
162  switch(op) {
163  case operation::deref:
164  if constexpr(is_meta_pointer_like_v<Type>) {
165  if constexpr(std::is_function_v<std::remove_const_t<typename std::pointer_traits<Type>::element_type>>) {
166  *static_cast<meta_any *>(other) = any_cast<Type>(value);
167  } else if constexpr(!std::is_same_v<std::remove_const_t<typename std::pointer_traits<Type>::element_type>, void>) {
168  using in_place_type = decltype(adl_meta_pointer_like<Type>::dereference(any_cast<const Type &>(value)));
169  static_cast<meta_any *>(other)->emplace<in_place_type>(adl_meta_pointer_like<Type>::dereference(any_cast<const Type &>(value)));
170  }
171  }
172  break;
173  case operation::seq:
174  if constexpr(is_complete_v<meta_sequence_container_traits<Type>>) {
175  *static_cast<meta_sequence_container *>(other) = {std::in_place_type<Type>, std::move(const_cast<any &>(value))};
176  }
177  break;
178  case operation::assoc:
179  if constexpr(is_complete_v<meta_associative_container_traits<Type>>) {
180  *static_cast<meta_associative_container *>(other) = {std::in_place_type<Type>, std::move(const_cast<any &>(value))};

183  }

980 class meta_type {
981  template<auto Member, typename Pred>
982  [[nodiscard]] const auto *lookup(meta_any *const args, const typename internal::meta_type_node::size_type sz, Pred pred) const {
983  std::decay_t<decltype(node->*Member)> candidate{};
984  size_type extent{sz + 1u};
985  bool ambiguous{};
986 
987  for(auto *curr = (node->*Member); curr; curr = curr->next) {
988  if(pred(curr) && curr->arity == sz) {
989  size_type direct{};
990  size_type ext{};
991 
992  for(size_type next{}; next < sz && next == (direct + ext) && args[next]; ++next) {
993  const auto type = args[next].type();
994  const auto other = curr->arg(next);
995 
996  if(const auto &info = other.info(); info == type.info()) {
997  ++direct;
998  } else {
999  ext += internal::find_by<&node_type::base>(info, type.node)
1000  || internal::find_by<&node_type::conv>(info, type.node)
1001  || (type.node->conversion_helper && other.node->conversion_helper);
1002  }
1003  }
1004 
1005  if((direct + ext) == sz) {
1006  if(ext < extent) {
1007  candidate = curr;
1008  extent = ext;
1009  ambiguous = false;
1010  } else if(ext == extent) {
1011  ambiguous = true;
1012  }
1013  }
1014  }
1015  }
1016
