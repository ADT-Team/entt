31 [[nodiscard]] bool valid() const noexcept {

34 && std::apply([entt = *it](const auto *...curr) { return (!curr->contains(entt) && ...); }, filter);
35 }

199 [[nodiscard]] auto opaque_check_set() const noexcept {
200 std::array<const base_type *, sizeof...(Get) - 1u> other{};
201 std::apply([&other, pos = 0u, view = view](const auto *...curr) mutable { ((curr == view ? void() : void(other[pos++] = curr)), ...); }, pools);

209 template<std::size_t Curr, std::size_t Other, typename... Args>
210 [[nodiscard]] auto dispatch_get(const std::tuple<underlying_type, Args...> &curr) const {
211 if constexpr(Curr == Other) {
212 return std::forward_as_tuple(std::get<Args>(curr)...);
213 } else {
214 return storage<Other>().get_as_tuple(std::get<0>(curr));
215 }

218 [[nodiscard]] auto reject(const underlying_type entt) const noexcept {

222 template<std::size_t Curr, typename Func, std::size_t... Index>
223 void each(Func &func, std::index_sequence<Index...>) const {
224 for(const auto curr: storage<Curr>().each()) {
225 if(const auto entt = std::get<0>(curr); ((sizeof...(Get) != 1u) || (entt != tombstone)) && ((Curr == Index || storage<Index>().contains(entt)) && ...) && !reject(entt)) {
226 if constexpr(is_applicable_v<Func, decltype(std::tuple_cat(std::tuple<entity_type>{}, std::declval<basic_view>().get({})))>) {
227 std::apply(func, std::tuple_cat(std::make_tuple(entt), dispatch_get<Curr, Index>(curr)...));
228 } else {
229 std::apply(func, std::tuple_cat(dispatch_get<Curr, Index>(curr)...));
230 }
231 }
232 }
233 }

235 template<typename Func, std::size_t... Index>
236 void pick_and_each(Func &func, std::index_sequence<Index...> seq) const {
237 ((&storage<Index>() == view ? each<Index>(func, seq) : void()), ...);
238 }
