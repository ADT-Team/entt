259  virtual void try_emplace(const Entity entt, const void * = nullptr) {
260  const auto pos = static_cast<size_type>(entity_traits::to_entity(entt));
261  const auto page = pos / entity_traits::page_size;
262 
263  if(!(page < sparse.size())) {
264  sparse.resize(page + 1u, nullptr);
265  }
266 
267  if(!sparse[page]) {
268  auto page_allocator{packed.get_allocator()};
269  sparse[page] = alloc_traits::allocate(page_allocator, entity_traits::page_size);
270  std::uninitialized_fill(sparse[page], sparse[page] + entity_traits::page_size, null);
271  }
272 
273  auto &elem = sparse[page][fast_mod(pos, entity_traits::page_size)];
274  ENTT_ASSERT(entity_traits::to_version(elem) == entity_traits::to_version(tombstone), "Slot not available");
275 
276  if(free_list == null) {
277  elem = entity_traits::combine(static_cast<typename entity_traits::entity_type>(packed.size()), entity_traits::to_integral(entt));
278  packed.push_back(entt);
279  } else {
280  elem = entity_traits::combine(entity_traits::to_integral(free_list), entity_traits::to_integral(entt));
281  free_list = std::exchange(packed[static_cast<size_type>(entity_traits::to_entity(free_list))], entt);
282  }

753  void compact() {
754  size_type next = packed.size();
755  for(; next && packed[next - 1u] == tombstone; --next) {}
756 
757  for(auto *it = &free_list; *it != null && next; it = std::addressof(packed[entity_traits::to_entity(*it)])) {
758  if(const size_type pos = entity_traits::to_entity(*it); pos < next) {
759  --next;
760  move_and_pop(next, pos);
761  std::swap(packed[next], packed[pos]);
762  const auto entity = static_cast<typename entity_traits::entity_type>(pos);
763  sparse_ref(packed[pos]) = entity_traits::combine(entity, entity_traits::to_integral(packed[pos]));
764  *it = entity_traits::combine(static_cast<typename entity_traits::entity_type>(next), entity_traits::reserved);
765  for(; next && packed[next - 1u] == tombstone; --next) {}
766  }
767  }
768 
769  free_list = tombstone;
770  packed.resize(next);
771  }

786  void swap_elements(const entity_type lhs, const entity_type rhs) {
787  ENTT_ASSERT(contains(lhs) && contains(rhs), "Set does not contain entities");
788 
789  auto &entt = sparse_ref(lhs);
790  auto &other = sparse_ref(rhs);
791 
792  const auto from = entity_traits::to_entity(entt);
793  const auto to = entity_traits::to_entity(other);
794 
795  // basic no-leak guarantee (with invalid state) if swapping throws
796  swap_at(static_cast<size_type>(from), static_cast<size_type>(to));
797  entt = entity_traits::combine(to, entity_traits::to_integral(packed[from]));
798  other = entity_traits::combine(from, entity_traits::to_integral(packed[to]));
799  std::swap(packed[from], packed[to]);

832  template<typename Compare, typename Sort = std_sort, typename... Args>
833  void sort_n(const size_type length, Compare compare, Sort algo = Sort{}, Args &&...args) {
834  ENTT_ASSERT(!(length > packed.size()), "Length exceeds the number of elements");
835  ENTT_ASSERT(free_list == null, "Partial sorting with tombstones is not supported");
836 
837  algo(packed.rend() - length, packed.rend(), std::move(compare), std::forward<Args>(args)...);
838 
839  for(size_type pos{}; pos < length; ++pos) {
840  auto curr = pos;
841  auto next = index(packed[curr]);
842 
843  while(curr != next) {
844  const auto idx = index(packed[next]);
845  const auto entt = packed[curr];
846 
847  swap_at(next, idx);
848  const auto entity = static_cast<typename entity_traits::entity_type>(curr);
849  sparse_ref(entt) = entity_traits::combine(entity, entity_traits::to_integral(packed[curr]));
850  curr = std::exchange(next, idx);
851  }
852  }
