174  [[nodiscard]] auto assure_page(const std::size_t idx) {
175  if(!(idx < bucket)) {
176  const size_type sz = idx + 1u;
177  const auto mem = bucket_alloc_traits::allocate(bucket_allocator, sz);
178 
179  std::uninitialized_value_construct(mem + bucket, mem + sz);
180  std::uninitialized_copy(sparse, sparse + bucket, mem);
181 
182  std::destroy(sparse, sparse + bucket);
183  bucket_alloc_traits::deallocate(bucket_allocator, sparse, bucket);
184

186  bucket = sz;
187  }
188

190  sparse[idx] = alloc_traits::allocate(allocator, sparse_page);
191  std::uninitialized_fill(sparse[idx], sparse[idx] + sparse_page, null);
192  }

688  void compact() {
689  size_type next = count;
690  for(; next && packed[next - 1u] == tombstone; --next);
691 
692  for(auto *it = &free_list; *it != null && next; it = std::addressof(packed[traits_type::to_entity(*it)])) {
693  if(const size_type pos = traits_type::to_entity(*it); pos < next) {
694  --next;
695  move_and_pop(next, pos);
696  std::swap(packed[next], packed[pos]);
697  sparse[page(packed[pos])][offset(packed[pos])] = traits_type::construct(static_cast<const typename traits_type::entity_type>(pos));
698  *it = traits_type::construct(static_cast<typename traits_type::entity_type>(next));
699  for(; next && packed[next - 1u] == tombstone; --next);
700  }
701  }
702 
703  free_list = tombstone;
704  count = next;
705  }

720  void swap(const entity_type lhs, const entity_type rhs) {
721  ENTT_ASSERT(contains(lhs), "Set does not contain entity");
722  ENTT_ASSERT(contains(rhs), "Set does not contain entity");
723 
724  auto &entt = sparse[page(lhs)][offset(lhs)];
725  auto &other = sparse[page(rhs)][offset(rhs)];
726 
727  const auto from = size_type{traits_type::to_entity(entt)};
728  const auto to = size_type{traits_type::to_entity(other)};
729 
730  // basic no-leak guarantee (with invalid state) if swapping throws
731  swap_at(from, to);
732  std::swap(entt, other);
733  std::swap(packed[from], packed[to]);
734  }

767  void sort_n(const size_type length, Compare compare, Sort algo = Sort{}, Args &&... args) {
768  // basic no-leak guarantee (with invalid state) if sorting throws
769  ENTT_ASSERT(!(length > count), "Length exceeds the number of elements");
770  compact();
771

775  auto curr = pos;

779  const auto idx = index(packed[next]);
780  const auto entt = packed[curr];

784  curr = std::exchange(next, idx);
