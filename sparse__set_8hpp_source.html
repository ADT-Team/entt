181 [[nodiscard]] auto sparse_ptr(const Entity entt) const {
182 const auto pos = static_cast<size_type>(entity_traits::to_entity(entt));
183 const auto page = pos / entity_traits::page_size;
184 return (page < sparse.size() && sparse[page]) ? (sparse[page] + fast_mod(pos, entity_traits::page_size)) : nullptr;
185 }
186
187 [[nodiscard]] auto &sparse_ref(const Entity entt) const {
188 ENTT_ASSERT(sparse_ptr(entt), "Invalid element");
189 const auto pos = static_cast<size_type>(entity_traits::to_entity(entt));
190 return sparse[pos / entity_traits::page_size][fast_mod(pos, entity_traits::page_size)];
191 }
192
193 [[nodiscard]] auto &assure_at_least(const Entity entt) {
194 const auto pos = static_cast<size_type>(entity_traits::to_entity(entt));
195 const auto page = pos / entity_traits::page_size;
196
197 if(!(page < sparse.size())) {
198 sparse.resize(page + 1u, nullptr);
199 }
200
201 if(!sparse[page]) {
202 auto page_allocator{packed.get_allocator()};
203 sparse[page] = alloc_traits::allocate(page_allocator, entity_traits::page_size);
204 std::uninitialized_fill(sparse[page], sparse[page] + entity_traits::page_size, null);
205 }
206
207 auto &elem = sparse[page][fast_mod(pos, entity_traits::page_size)];
208 ENTT_ASSERT(elem == null, "Slot not available");
209 return elem;
210 }
211
212 void release_sparse_pages() {
213 auto page_allocator{packed.get_allocator()};
214
215 for(auto &&page: sparse) {
216 if(page != nullptr) {
217 std::destroy(page, page + entity_traits::page_size);
218 alloc_traits::deallocate(page_allocator, page, entity_traits::page_size);
219 page = nullptr;
220 }
221 }
222 }

240 void swap_and_pop(const basic_iterator it) {
241 ENTT_ASSERT(mode == deletion_policy::swap_and_pop, "Deletion policy mismatched");
242 auto &self = sparse_ref(*it);
243 const auto entt = entity_traits::to_entity(self);
244 sparse_ref(packed.back()) = entity_traits::combine(entt, entity_traits::to_integral(packed.back()));
245 packed[static_cast<size_type>(entt)] = packed.back();
246 // unnecessary but it helps to detect nasty bugs
247 ENTT_ASSERT((packed.back() = null, true), "");
248 // lazy self-assignment guard
249 self = null;
250 packed.pop_back();
251 }

257 void in_place_pop(const basic_iterator it) {
258 ENTT_ASSERT(mode == deletion_policy::in_place, "Deletion policy mismatched");
259 const auto entt = entity_traits::to_entity(std::exchange(sparse_ref(*it), null));
260 packed[static_cast<size_type>(entt)] = std::exchange(free_list, entity_traits::combine(entt, entity_traits::reserved));
261 }

287 virtual basic_iterator try_emplace(const Entity entt, const bool force_back, const void * = nullptr) {
288 ENTT_ASSERT(!contains(entt), "Set already contains entity");
289
290 if(auto &elem = assure_at_least(entt); free_list == null || force_back) {
291 packed.push_back(entt);
292 elem = entity_traits::combine(static_cast<typename entity_traits::entity_type>(packed.size() - 1u), entity_traits::to_integral(entt));
293 return begin();
294 } else {
295 const auto pos = static_cast<size_type>(entity_traits::to_entity(free_list));
296 elem = entity_traits::combine(entity_traits::to_integral(free_list), entity_traits::to_integral(entt));
297 free_list = std::exchange(packed[pos], entt);
298 return --(end() - pos);
299 }
300 }

780 void compact() {
781 size_type from = packed.size();
782 for(; from && packed[from - 1u] == tombstone; --from) {}
783
784 for(auto *it = &free_list; *it != null && from; it = std::addressof(packed[entity_traits::to_entity(*it)])) {
785 if(const size_type to = entity_traits::to_entity(*it); to < from) {
786 --from;
787 move_element(from, to);
788
789 using std::swap;
790 swap(packed[from], packed[to]);
791
792 const auto entity = static_cast<typename entity_traits::entity_type>(to);
793 sparse_ref(packed[to]) = entity_traits::combine(entity, entity_traits::to_integral(packed[to]));
794 *it = entity_traits::combine(static_cast<typename entity_traits::entity_type>(from), entity_traits::reserved);
795 for(; from && packed[from - 1u] == tombstone; --from) {}
796 }
797 }
798
799 free_list = tombstone;
800 packed.resize(from);
801 }

816 void swap_elements(const entity_type lhs, const entity_type rhs) {
817 ENTT_ASSERT(contains(lhs) && contains(rhs), "Set does not contain entities");
818
819 auto &entt = sparse_ref(lhs);
820 auto &other = sparse_ref(rhs);
821
822 const auto from = entity_traits::to_entity(entt);
823 const auto to = entity_traits::to_entity(other);
824
825 // basic no-leak guarantee (with invalid state) if swapping throws
826 swap_at(static_cast<size_type>(from), static_cast<size_type>(to));
827 entt = entity_traits::combine(to, entity_traits::to_integral(packed[from]));
828 other = entity_traits::combine(from, entity_traits::to_integral(packed[to]));
829
830 using std::swap;
831 swap(packed[from], packed[to]);
832 }

864 template<typename Compare, typename Sort = std_sort, typename... Args>
865 void sort_n(const size_type length, Compare compare, Sort algo = Sort{}, Args &&...args) {
866 ENTT_ASSERT(!(length > packed.size()), "Length exceeds the number of elements");
867 ENTT_ASSERT(free_list == null, "Partial sorting with tombstones is not supported");
868
869 algo(packed.rend() - length, packed.rend(), std::move(compare), std::forward<Args>(args)...);
870
871 for(size_type pos{}; pos < length; ++pos) {
872 auto curr = pos;
873 auto next = index(packed[curr]);
874
875 while(curr != next) {
876 const auto idx = index(packed[next]);
877 const auto entt = packed[curr];
878
879 swap_at(next, idx);
880 const auto entity = static_cast<typename entity_traits::entity_type>(curr);
881 sparse_ref(entt) = entity_traits::combine(entity, entity_traits::to_integral(packed[curr]));
882 curr = std::exchange(next, idx);
883 }
884 }
885 }

920 void respect(const basic_sparse_set &other) {
921 compact();
922
923 const auto to = other.end();
924 auto from = other.begin();
925
926 for(size_type pos = packed.size() - 1; pos && from != to; ++from) {
927 if(contains(*from)) {
928 if(*from != packed[pos]) {
