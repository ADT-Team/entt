83  template<typename Type, auto Candidate, typename Ret, typename Any, typename... Args>
84  static void fill_vtable_entry(Ret(* &entry)(Any &, Args...)) {
85  if constexpr(std::is_invocable_r_v<Ret, decltype(Candidate), Args...>) {
86  entry = +[](Any &, Args... args) -> Ret {
