266 [[nodiscard]] auto insert_or_do_nothing(Other &&value) {
267 const auto index = value_to_bucket(value);
268
269 if(auto it = constrained_find(value, index); it != end()) {
270 return std::make_pair(it, false);

273 packed.first().emplace_back(sparse.first()[index], std::forward<Other>(value));
274 sparse.first()[index] = packed.first().size() - 1u;

277 return std::make_pair(--end(), true);

280 void move_and_pop(const std::size_t pos) {
281 if(const auto last = size() - 1u; pos != last) {
282 size_type *curr = sparse.first().data() + value_to_bucket(packed.first().back().second);
283 packed.first()[pos] = std::move(packed.first().back());
284 for(; *curr != last; curr = &packed.first()[*curr].first) {}
285 *curr = pos;
286 }
287
288 packed.first().pop_back();
289 }

292 if(size() > (bucket_count() * max_load_factor())) {

530 std::pair<iterator, bool> emplace(Args &&...args) {
531 if constexpr(((sizeof...(Args) == 1u) && ... && std::is_same_v<std::decay_t<Args>, value_type>)) {
532 return insert_or_do_nothing(std::forward<Args>(args)...);

842 void rehash(const size_type cnt) {
843 auto value = cnt > minimum_capacity ? cnt : minimum_capacity;
844 const auto cap = static_cast<size_type>(size() / max_load_factor());
845 value = value > cap ? value : cap;
846
847 if(const auto sz = next_power_of_two(value); sz != bucket_count()) {
848 sparse.first().resize(sz);
849
850 for(auto &&elem: sparse.first()) {
851 elem = std::numeric_limits<size_type>::max();
852 }
853
854 for(size_type pos{}, last = size(); pos < last; ++pos) {
855 const auto index = value_to_bucket(packed.first()[pos].second);
856 packed.first()[pos].first = std::exchange(sparse.first()[index], pos);
857 }
858 }
