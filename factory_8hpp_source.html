58inline meta_func_node &meta_extend(internal::meta_type_node &parent, const id_type id, meta_func_node node) {
59 if(auto it = parent.details->func.find(id); it != parent.details->func.end()) {
60 for(auto *curr = &it->second; curr; curr = curr->next.get()) {
61 if(curr->invoke == node.invoke) {
62 node.next = std::move(curr->next);
63 *curr = std::move(node);
64 return *curr;
65 }
66 }
67
68 // locally overloaded function
69 node.next = std::make_shared<meta_func_node>(std::move(parent.details->func[id]));
70 }
71
72 return parent.details->func.insert_or_assign(id, std::move(node)).first->second;
73}

493 template<typename... Value>
494 meta_factory prop(id_type id, [[maybe_unused]] Value &&...value) {
495 ENTT_ASSERT(bucket != nullptr, "Meta object does not support properties");
496
497 if constexpr(sizeof...(Value) == 0u) {
498 internal::meta_extend(
499 *bucket,
500 id,
501 internal::meta_prop_node{
502 &internal::resolve<void>});

504 internal::meta_extend(
505 *bucket,
506 id,
507 internal::meta_prop_node{
508 &internal::resolve<std::decay_t<Value>>...,
