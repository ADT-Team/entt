278  template<typename... Args>
279  [[nodiscard]] auto get_or_emplace(const Key &key, Args &&...args) {
280  const auto hash = sparse.second()(key);
281  auto index = hash_to_bucket(hash);
282 
283  if(auto it = constrained_find(key, index); it != end()) {
284  return std::make_pair(it, false);
285  }
286 
287  if(const auto count = size() + 1u; count > (bucket_count() * max_load_factor())) {
288  rehash(bucket_count() * 2u);
289  index = hash_to_bucket(hash);
290  }
291 
292  packed.first().emplace_back(sparse.first()[index], std::forward<Args>(args)...);
293  // update goes after emplace to enforce exception guarantees
294  sparse.first()[index] = size() - 1u;
295 
296  return std::make_pair(--end(), true);
297  }

299  void move_and_pop(const std::size_t pos) {
300  if(const auto last = size() - 1u; pos != last) {
301  size_type *curr = sparse.first().data() + bucket(packed.first().back().element.first);
302  for(; *curr != last; curr = &packed.first()[*curr].next) {}
303  *curr = pos;
304 
305  using node_allocator_traits = typename alloc_traits::template rebind_traits<decltype(node_type::element)>;
306  typename node_allocator_traits::allocator_type allocator = packed.first().get_allocator();
307  auto *ptr = std::addressof(packed.first()[pos].element);
308 
309  std::destroy_at(ptr);
310  packed.first()[pos].next = packed.first().back().next;
311  // no exception guarantees when mapped type has a throwing move constructor (we're technically doomed)
312  node_allocator_traits::construct(allocator, ptr, std::move(packed.first().back().element));
313  }
314 
315  packed.first().pop_back();
316  }

922  }
923  }

932  rehash(static_cast<size_type>(std::ceil(count / max_load_factor())));
933  }
