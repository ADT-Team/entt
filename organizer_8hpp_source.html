162  [[nodiscard]] std::vector<bool> adjacency_matrix() {
163  const auto length = vertices.size();
164  std::vector<bool> edges(length * length, false);
165 
166  // creates the ajacency matrix
167  for(const auto &deps: dependencies) {
168  const auto last = deps.second.cend();
169  auto it = deps.second.cbegin();
170 
171  while(it != last) {
172  if(it->second) {
173  // rw item
174  if(auto curr = it++; it != last) {
175  if(it->second) {
176  edges[curr->first * length + it->first] = true;
177  } else {
178  if(const auto next = std::find_if(it, last, [](const auto &elem) { return elem.second; }); next != last) {
179  for(; it != next; ++it) {
180  edges[curr->first * length + it->first] = true;
181  edges[it->first * length + next->first] = true;
182  }
183  } else {
184  for(; it != next; ++it) {
185  edges[curr->first * length + it->first] = true;
186  }
187  }
188  }
189  }
190  } else {
191  // ro item, possibly only on first iteration
192  if(const auto next = std::find_if(it, last, [](const auto &elem) { return elem.second; }); next != last) {
193  for(; it != next; ++it) {
194  edges[it->first * length + next->first] = true;
195  }
196  } else {
197  it = last;
198  }
199  }
200  }
201  }
202 
203  // computes the transitive closure
204  for(std::size_t vk{}; vk < length; ++vk) {
205  for(std::size_t vi{}; vi < length; ++vi) {
206  for(std::size_t vj{}; vj < length; ++vj) {
207  edges[vi * length + vj] = edges[vi * length + vj] || (edges[vi * length + vk] && edges[vk * length + vj]);
208  }
209  }
210  }
211 
212  // applies the transitive reduction
213  for(std::size_t vert{}; vert < length; ++vert) {
214  edges[vert * length + vert] = false;
215  }
216 
217  for(std::size_t vj{}; vj < length; ++vj) {
218  for(std::size_t vi{}; vi < length; ++vi) {
219  if(edges[vi * length + vj]) {
220  for(std::size_t vk{}; vk < length; ++vk) {
221  if(edges[vj * length + vk]) {
222  edges[vi * length + vk] = false;
223  }
224  }
225  }
226  }
227  }
228 
229  return edges;
230  }

446  std::vector<vertex> graph() {
447  const auto edges = adjacency_matrix();
448 
449  // creates the adjacency list
450  std::vector<vertex> adjacency_list{};
451  adjacency_list.reserve(vertices.size());
452 
453  for(std::size_t col{}, length = vertices.size(); col < length; ++col) {
454  std::vector<std::size_t> reachable{};
455  const auto row = col * length;
456  bool is_top_level = true;
457 
458  for(std::size_t next{}; next < length; ++next) {
459  if(edges[row + next]) {
460  reachable.push_back(next);
461  }
462  }
463 
464  for(std::size_t next{}; next < length && is_top_level; ++next) {
465  is_top_level = !edges[next * length + col];
466  }
467 
468  adjacency_list.emplace_back(is_top_level, vertices[col], std::move(reachable));
469  }
470 
471  return adjacency_list;
472  }
