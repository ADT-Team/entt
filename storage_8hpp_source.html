250  auto assure_at_least(const std::size_t pos) {
251  auto &&container = packed.first();
252  const auto idx = pos / comp_traits::page_size;
253 
254  if(!(idx < container.size())) {
255  auto curr = container.size();
256  container.resize(idx + 1u, nullptr);
257 
258  ENTT_TRY {
259  for(const auto last = container.size(); curr < last; ++curr) {
260  container[curr] = alloc_traits::allocate(packed.second(), comp_traits::page_size);
261  }
262  }
263  ENTT_CATCH {
264  container.resize(curr);
265  ENTT_THROW;
266  }
267  }
268 
269  return container[idx] + fast_mod(pos, comp_traits::page_size);
