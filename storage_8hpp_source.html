193  void assure_at_least(const std::size_t last) {
194  if(const auto idx = page(last - 1u); !(idx < bucket)) {
195  const size_type sz = idx + 1u;
196  const auto mem = bucket_alloc_traits::allocate(bucket_allocator, sz);
197  std::uninitialized_copy(packed, packed + bucket, mem);
198  size_type pos{};
199 
200  ENTT_TRY {
201  for(pos = bucket; pos < sz; ++pos) {
202  auto pg = alloc_traits::allocate(allocator, packed_page);
203  bucket_alloc_traits::construct(bucket_allocator, std::addressof(mem[pos]), pg);
204  }
205  } ENTT_CATCH {
206  for(auto next = bucket; next < pos; ++next) {
207  alloc_traits::deallocate(allocator, mem[next], packed_page);
208  }
209

211  bucket_alloc_traits::deallocate(bucket_allocator, mem, sz);
212  ENTT_THROW;
213  }
214 
215  std::destroy(packed, packed + bucket);
216  bucket_alloc_traits::deallocate(bucket_allocator, packed, bucket);
217

219  bucket = sz;
220  }
