331  template<typename Container>
332  auto update(int, Container &container)
333  -> decltype(typename Container::mapped_type{}, void()) {
334  // map like container
335  Container other;
336 
337  for(auto &&pair: container) {
338  using first_type = std::remove_const_t<typename std::decay_t<decltype(pair)>::first_type>;
339  using second_type = typename std::decay_t<decltype(pair)>::second_type;
340 
341  if constexpr(std::is_same_v<first_type, entity_type> && std::is_same_v<second_type, entity_type>) {
342  other.emplace(map(pair.first), map(pair.second));
343  } else if constexpr(std::is_same_v<first_type, entity_type>) {
344  other.emplace(map(pair.first), std::move(pair.second));
345  } else {
346  static_assert(std::is_same_v<second_type, entity_type>, "Neither the key nor the value are of entity type");
347  other.emplace(std::move(pair.first), map(pair.second));
348  }
349  }
350 
351  std::swap(container, other);
352  }
