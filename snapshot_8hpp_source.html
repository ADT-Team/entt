323  template<typename Container>
324  auto update(int, Container &container)
325  -> decltype(typename Container::mapped_type{}, void()) {
326  // map like container
327  Container other;
328 
329  for(auto &&pair: container) {
330  using first_type = std::remove_const_t<typename std::decay_t<decltype(pair)>::first_type>;
331  using second_type = typename std::decay_t<decltype(pair)>::second_type;
332 
333  if constexpr(std::is_same_v<first_type, entity_type> && std::is_same_v<second_type, entity_type>) {
334  other.emplace(map(pair.first), map(pair.second));
335  } else if constexpr(std::is_same_v<first_type, entity_type>) {
336  other.emplace(map(pair.first), std::move(pair.second));
337  } else {
338  static_assert(std::is_same_v<second_type, entity_type>, "Neither the key nor the value are of entity type");
339  other.emplace(std::move(pair.first), map(pair.second));
340  }
341  }
342 
343  std::swap(container, other);
344  }
